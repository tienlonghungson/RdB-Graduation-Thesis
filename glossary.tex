\newglossaryentry{lfsr}{
    type=\acronymtype,
    name={LFSR},
    description={Linear feedback shift register: an algorithm to generate a de Bruijn sequence using a linear function},
    first = {Linear feedback shift register}
}

\newglossaryentry{fkm}{
    type=\acronymtype,
    name={FKM},
    description={Algorithm of Fredricksen, Kessler and Maiorana to generate lexicographically all Lyndon words of length being divisor of $n$},
}

\newglossaryentry{qkd}{
    type=\acronymtype,
    name={QKD},
    description={Quantum key distribution: a secure communication method involving components of quantum mechanics for exchanging encryption keys only known between shared parties}
    first = {Quantum key distribution}
}

\newglossaryentry{HdB}{
    type=\acronymtype,
    name={HdB},
    description={Hybrid de Bruijn sequence: de Bruijn sequences encoded with a beacon model, on-on is $1$ and on-off is $0$}
    first = {Hybrid de Bruijn sequence}
}

\newglossaryentry{dBTS}{
    type=\acronymtype,
    name={dBTS},
    description={de Bruijn based Timing and Synchronization system: a timing and synchronization system using Hybrid de Bruijn code proposed by \cite{zhang2021timing}}
    first = {de Bruijn based timing and synchronization system}
}

\newglossaryentry{RdB}{
    type =\acronymtype,
    name={RdB},
    description={Run length limited de Bruijn sequence: a sequence that is the combination of positioning sequence and run length sequnce}
    first = {Run length limited de Bruijn sequence}
}

\newglossaryentry{RLL}{
    type =\acronymtype,
    name ={RLL},
    description={Run length limited: a line coding technique that is used to send arbitrary data over a communications channel with bandwidth limits}
